1. 스택/큐

- 프린터

- 큐를 이용

- 조건

  ```
  1. 인쇄 대기목록의 가장 앞에 있는 문서(J)를 대기목록에서 꺼냅니다.
  2. 나머지 인쇄 대기목록에서 J보다 중요도가 높은 문서가 한 개라도 존재하면 J를 대기목록의 가장 마지막에 넣습니다.
  3. 그렇지 않으면 J를 인쇄합니다.
  ```

```python
from collections import deque

def solution(priorities, location):
    answer = 0
    q = deque()
    count = 1
    
    
    # 인덱스와 중요도를 큐에 넣는다.
    for i in range(len(priorities)):
        q.append((i,priorities[i]))
    

    while q:
        copy = True
        index, priority = q.popleft()
        
        
        for que in q:
            i,p = que
            # 만약 현재 중요도보다 높은 중요도가 큐에 있다면
            if p > priority:
                copy = False
                # 현재 문서를 다시 큐에 넣어준다.
                q.append((index,priority))
                break
        
        # 현재 문서보다 중요도가 높은 문서가 없다면
        if copy == True:
            # 찾고 있는 인덱스를 찾았을 때 몇번째로 나왔는지를 알 수 있는 count를 answer에 넣어준다.
            if index == location:
                answer = count
                break
            count+=1
            
    return answer
```

3. DFS/BFS

- 단어 변환
- DFS 혹은 BFS를 이용해서 단어가 하나만 다른 것들을 스택이나 큐에 깊이와 같이 넣는다.
- stack.append((words[i], depth + 1))
- visited 이용

```python
from collections import deque


def solution(begin, target, words):

    answer = 0
    # 스택이나 큐를 이용한다.
    # stack = [(begin, 0)]
    stack = deque()
    stack.append((begin,0))
    visited = [False] * len(words)

    while stack:
        # now, depth = stack.pop()
        now, depth = stack.popleft()

        print('now = ', now)
		
        # target과 일치하면 깊이 반환
        if now == target:
            answer = depth
            break

        # for문을 돌면서 현재 단어와 알파벳이 하나 차이나는 것들을 스택 or 큐에 넣어준다.
        for i in range(len(words)):
            if visited[i] == True:
                continue

            count = 0
            for j in range(len(target)):
                if words[i][j] == now[j]:
                    count += 1
            if count == len(target) - 1:
                # 깊이는 스택에 같이 넣어준다!
                stack.append((words[i],depth+1))
                visited[i] = True
                print(words[i], depth+1)

    return answer
```

def solution(numbers):
    # 숫자를 문자열로 바꿔서 저장

5. 그래프

- 가장 먼 노드

```python
import heapq

def solution(n, edge):
    answer = 0

    graph = [[] for i in range(n+1)]
    # 노드 1에서 다른 노드까지의 거리 구하기
    start = 1
    # 처음은 1에서 모든 노드들까지의 거리 무한으로 설정
    distance = [1e9] * (n+1)

    
    # 간선 정보를 입력받아 노드, 거리로 append하기
    for i in edge:
        a = i[0]
        b = i[1]
        graph[a].append((b,1))
        graph[b].append((a,1))

    
    q = []
    # 힙을 사용하기 위해 (거리, 노드 번호) 순으로 push
    heapq.heappush(q,(0,start))
    distance[start] = 0

    while q:
        # 우선순위 큐로 가장 거리가 작은 노드부터 pop
        dist, now = heapq.heappop(q)
        # 이미 distance가 갱신되었으면 continue
        if distance[now] < dist:
            continue

        # 현재 노드와 연결된 노드들 살펴보기
        for i in graph[now]:
            cost = dist + i[1]
            # 노드 번호 i[0]까지의 거리가 갱신될 수 있으면 갱신하고, heap에 push하기
            if cost < distance[i[0]]:
                distance[i[0]] = cost
                heapq.heappush(q,(cost,i[0]))


    # 최대 거리를 가진 노드의 개수 세기
    for i in range(1,len(distance)):
        if distance[i] == max(distance[1:]):
            answer+=1



    return answer
```
